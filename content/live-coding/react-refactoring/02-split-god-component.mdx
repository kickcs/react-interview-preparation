title: "Разбить God-компонент Dashboard"

---

Компонент `Dashboard` делает слишком много: рендерит фильтры, таблицу данных и пагинацию — всё в одном месте. При этом логика перемешана с разметкой, а любое изменение затрагивает весь файл. Разбейте его на три самостоятельных компонента: `FilterBar`, `DataTable` и `Pagination`.

<StarterCode>

```jsx
import { useState } from 'react';

const USERS = [
  { id: 1, name: 'Анна Петрова', role: 'admin', status: 'active' },
  { id: 2, name: 'Борис Иванов', role: 'user', status: 'inactive' },
  { id: 3, name: 'Вера Сидорова', role: 'user', status: 'active' },
  { id: 4, name: 'Григорий Козлов', role: 'moderator', status: 'active' },
  { id: 5, name: 'Дарья Новикова', role: 'user', status: 'inactive' },
  { id: 6, name: 'Евгений Морозов', role: 'admin', status: 'active' },
];

const PAGE_SIZE = 2;

export default function Dashboard() {
  const [search, setSearch] = useState('');
  const [roleFilter, setRoleFilter] = useState('all');
  const [statusFilter, setStatusFilter] = useState('all');
  const [page, setPage] = useState(1);

  const filtered = USERS.filter(u => {
    const matchesSearch = u.name.toLowerCase().includes(search.toLowerCase());
    const matchesRole = roleFilter === 'all' || u.role === roleFilter;
    const matchesStatus = statusFilter === 'all' || u.status === statusFilter;
    return matchesSearch && matchesRole && matchesStatus;
  });

  const totalPages = Math.ceil(filtered.length / PAGE_SIZE);
  const paginated = filtered.slice((page - 1) * PAGE_SIZE, page * PAGE_SIZE);

  return (
    <div style={{ padding: 24 }}>
      <h1>Пользователи</h1>

      {/* Фильтры */}
      <div style={{ display: 'flex', gap: 12, marginBottom: 16 }}>
        <input
          placeholder="Поиск по имени..."
          value={search}
          onChange={e => { setSearch(e.target.value); setPage(1); }}
        />
        <select value={roleFilter} onChange={e => { setRoleFilter(e.target.value); setPage(1); }}>
          <option value="all">Все роли</option>
          <option value="admin">Admin</option>
          <option value="moderator">Moderator</option>
          <option value="user">User</option>
        </select>
        <select value={statusFilter} onChange={e => { setStatusFilter(e.target.value); setPage(1); }}>
          <option value="all">Все статусы</option>
          <option value="active">Active</option>
          <option value="inactive">Inactive</option>
        </select>
      </div>

      {/* Таблица */}
      <table style={{ width: '100%', borderCollapse: 'collapse' }}>
        <thead>
          <tr>
            <th style={{ textAlign: 'left', padding: 8, borderBottom: '2px solid #ddd' }}>Имя</th>
            <th style={{ textAlign: 'left', padding: 8, borderBottom: '2px solid #ddd' }}>Роль</th>
            <th style={{ textAlign: 'left', padding: 8, borderBottom: '2px solid #ddd' }}>Статус</th>
          </tr>
        </thead>
        <tbody>
          {paginated.map(user => (
            <tr key={user.id}>
              <td style={{ padding: 8, borderBottom: '1px solid #eee' }}>{user.name}</td>
              <td style={{ padding: 8, borderBottom: '1px solid #eee' }}>{user.role}</td>
              <td style={{ padding: 8, borderBottom: '1px solid #eee' }}>
                <span style={{ color: user.status === 'active' ? 'green' : 'gray' }}>
                  {user.status}
                </span>
              </td>
            </tr>
          ))}
          {paginated.length === 0 && (
            <tr><td colSpan={3} style={{ padding: 8, textAlign: 'center' }}>Ничего не найдено</td></tr>
          )}
        </tbody>
      </table>

      {/* Пагинация */}
      <div style={{ display: 'flex', gap: 8, marginTop: 16, alignItems: 'center' }}>
        <button onClick={() => setPage(p => p - 1)} disabled={page === 1}>←</button>
        {Array.from({ length: totalPages }, (_, i) => (
          <button
            key={i + 1}
            onClick={() => setPage(i + 1)}
            style={{ fontWeight: page === i + 1 ? 'bold' : 'normal' }}
          >
            {i + 1}
          </button>
        ))}
        <button onClick={() => setPage(p => p + 1)} disabled={page === totalPages}>→</button>
        <span>Страница {page} из {totalPages || 1}</span>
      </div>
    </div>
  );
}
```

</StarterCode>

<Hint title="Подсказка 1">
Начните с определения пропсов для каждого нового компонента. `FilterBar` получает значения фильтров и коллбэки на изменение. `DataTable` — массив пользователей для отображения. `Pagination` — текущую страницу, общее количество страниц и коллбэк `onPageChange`.
</Hint>

<Hint title="Подсказка 2">
Весь `state` (`search`, `roleFilter`, `statusFilter`, `page`) и логика фильтрации остаются в `Dashboard`. Дочерние компоненты — "глупые": они только отображают данные и вызывают переданные коллбэки. Не забудьте сбрасывать `page` в 1 при смене фильтров.
</Hint>

<Solution>

```jsx
import { useState } from 'react';

const USERS = [
  { id: 1, name: 'Анна Петрова', role: 'admin', status: 'active' },
  { id: 2, name: 'Борис Иванов', role: 'user', status: 'inactive' },
  { id: 3, name: 'Вера Сидорова', role: 'user', status: 'active' },
  { id: 4, name: 'Григорий Козлов', role: 'moderator', status: 'active' },
  { id: 5, name: 'Дарья Новикова', role: 'user', status: 'inactive' },
  { id: 6, name: 'Евгений Морозов', role: 'admin', status: 'active' },
];

const PAGE_SIZE = 2;

function FilterBar({ search, role, status, onSearchChange, onRoleChange, onStatusChange }) {
  return (
    <div style={{ display: 'flex', gap: 12, marginBottom: 16 }}>
      <input
        placeholder="Поиск по имени..."
        value={search}
        onChange={e => onSearchChange(e.target.value)}
      />
      <select value={role} onChange={e => onRoleChange(e.target.value)}>
        <option value="all">Все роли</option>
        <option value="admin">Admin</option>
        <option value="moderator">Moderator</option>
        <option value="user">User</option>
      </select>
      <select value={status} onChange={e => onStatusChange(e.target.value)}>
        <option value="all">Все статусы</option>
        <option value="active">Active</option>
        <option value="inactive">Inactive</option>
      </select>
    </div>
  );
}

function DataTable({ users }) {
  return (
    <table style={{ width: '100%', borderCollapse: 'collapse' }}>
      <thead>
        <tr>
          <th style={{ textAlign: 'left', padding: 8, borderBottom: '2px solid #ddd' }}>Имя</th>
          <th style={{ textAlign: 'left', padding: 8, borderBottom: '2px solid #ddd' }}>Роль</th>
          <th style={{ textAlign: 'left', padding: 8, borderBottom: '2px solid #ddd' }}>Статус</th>
        </tr>
      </thead>
      <tbody>
        {users.map(user => (
          <tr key={user.id}>
            <td style={{ padding: 8, borderBottom: '1px solid #eee' }}>{user.name}</td>
            <td style={{ padding: 8, borderBottom: '1px solid #eee' }}>{user.role}</td>
            <td style={{ padding: 8, borderBottom: '1px solid #eee' }}>
              <span style={{ color: user.status === 'active' ? 'green' : 'gray' }}>
                {user.status}
              </span>
            </td>
          </tr>
        ))}
        {users.length === 0 && (
          <tr><td colSpan={3} style={{ padding: 8, textAlign: 'center' }}>Ничего не найдено</td></tr>
        )}
      </tbody>
    </table>
  );
}

function Pagination({ page, totalPages, onPageChange }) {
  return (
    <div style={{ display: 'flex', gap: 8, marginTop: 16, alignItems: 'center' }}>
      <button onClick={() => onPageChange(page - 1)} disabled={page === 1}>←</button>
      {Array.from({ length: totalPages }, (_, i) => (
        <button
          key={i + 1}
          onClick={() => onPageChange(i + 1)}
          style={{ fontWeight: page === i + 1 ? 'bold' : 'normal' }}
        >
          {i + 1}
        </button>
      ))}
      <button onClick={() => onPageChange(page + 1)} disabled={page === totalPages}>→</button>
      <span>Страница {page} из {totalPages || 1}</span>
    </div>
  );
}

export default function Dashboard() {
  const [search, setSearch] = useState('');
  const [roleFilter, setRoleFilter] = useState('all');
  const [statusFilter, setStatusFilter] = useState('all');
  const [page, setPage] = useState(1);

  const filtered = USERS.filter(u => {
    const matchesSearch = u.name.toLowerCase().includes(search.toLowerCase());
    const matchesRole = roleFilter === 'all' || u.role === roleFilter;
    const matchesStatus = statusFilter === 'all' || u.status === statusFilter;
    return matchesSearch && matchesRole && matchesStatus;
  });

  const totalPages = Math.ceil(filtered.length / PAGE_SIZE);
  const paginated = filtered.slice((page - 1) * PAGE_SIZE, page * PAGE_SIZE);

  const handleFilterChange = (setter) => (value) => {
    setter(value);
    setPage(1);
  };

  return (
    <div style={{ padding: 24 }}>
      <h1>Пользователи</h1>
      <FilterBar
        search={search}
        role={roleFilter}
        status={statusFilter}
        onSearchChange={handleFilterChange(setSearch)}
        onRoleChange={handleFilterChange(setRoleFilter)}
        onStatusChange={handleFilterChange(setStatusFilter)}
      />
      <DataTable users={paginated} />
      <Pagination page={page} totalPages={totalPages} onPageChange={setPage} />
    </div>
  );
}
```
