---
title: "Кастомный хук useToggle"
order: 1
---

Напишите кастомный хук `useToggle(initialValue)`, который возвращает массив `[value, toggle]`. Вызов `toggle` переключает булево значение на противоположное. Используйте хук в компоненте с кнопкой.

<StarterCode>

```jsx
import { useState } from 'react';

function useToggle(initialValue = false) {
  // Ваш код здесь
}

// ✅ Проверка — запустите в своём React-проекте
export default function App() {
  const [isOn, toggle] = useToggle(false);

  return (
    <div>
      <p>Состояние: {isOn ? 'ON' : 'OFF'}</p>
      <button onClick={toggle}>Переключить</button>
    </div>
  );
}
```

</StarterCode>

<Hint title="Подсказка 1">
Используйте `useState` внутри хука для хранения булевого значения.
</Hint>

<Hint title="Подсказка 2">
Функцию `toggle` реализуйте через функциональное обновление: `setState(prev => !prev)`. Это гарантирует корректное значение даже при пакетных обновлениях.
</Hint>

<Solution>

```jsx
import { useState, useCallback } from 'react';

function useToggle(initialValue = false) {
  const [value, setValue] = useState(initialValue);
  const toggle = useCallback(() => setValue(prev => !prev), []);
  return [value, toggle];
}

export default function App() {
  const [isOn, toggle] = useToggle(false);

  return (
    <div>
      <p>Состояние: {isOn ? 'ON' : 'OFF'}</p>
      <button onClick={toggle}>Переключить</button>
    </div>
  );
}
```

</Solution>
