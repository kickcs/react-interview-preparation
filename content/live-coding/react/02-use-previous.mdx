order: 2
---

Напишите кастомный хук `usePrevious(value)`, который возвращает **предыдущее** значение переменной. При первом рендере возвращает `undefined`. Используйте `useRef` для сохранения значения между рендерами.

<StarterCode>

```jsx
import { useState, useEffect, useRef } from 'react';

function usePrevious(value) {
  // Ваш код здесь
}

// ✅ Проверка — запустите в своём React-проекте
export default function App() {
  const [count, setCount] = useState(0);
  const prevCount = usePrevious(count);

  return (
    <div>
      <p>Текущее: {count}</p>
      <p>Предыдущее: {prevCount ?? 'нет'}</p>
      <button onClick={() => setCount(c => c + 1)}>+1</button>
    </div>
  );
}
```

</StarterCode>

<Hint title="Подсказка 1">
`useRef` не вызывает ре-рендер при изменении. Храните в нём значение с прошлого рендера — это именно то, что нужно.
</Hint>

<Hint title="Подсказка 2">
Используйте `useEffect` без зависимостей от `value` для обновления рефа **после** рендера. Важно: обновление должно происходить *после* того, как новое значение уже отображено, поэтому `useEffect`, а не тело функции.
</Hint>

<Solution>

```jsx
import { useState, useEffect, useRef } from 'react';

function usePrevious(value) {
  const ref = useRef(undefined);

  useEffect(() => {
    ref.current = value;
  }, [value]);

  return ref.current;
}

export default function App() {
  const [count, setCount] = useState(0);
  const prevCount = usePrevious(count);

  return (
    <div>
      <p>Текущее: {count}</p>
      <p>Предыдущее: {prevCount ?? 'нет'}</p>
      <button onClick={() => setCount(c => c + 1)}>+1</button>
    </div>
  );
}
```

</Solution>
